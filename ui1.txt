                    <div className="mb-1 flex items-center gap-2">
                      <span className="h-2 w-2 shrink-0 rounded-full bg-[#f59e0b] shadow-[0_0_20px_rgba(245,158,11,0.75)]" />
                      <h3 className="truncate text-sm font-black text-white sm:text-base">{isRtl ? currentEpisode.titleFa : currentEpisode.titleEn}</h3>
                    </div>
                    <p className="line-clamp-1 text-xs font-medium text-slate-500">{isRtl ? currentEpisode.descFa : currentEpisode.descEn}</p>
                  </div>
                </div>

                <button
                  type="button"
                  onClick={togglePlay}
                  className="mx-auto flex h-14 w-14 shrink-0 items-center justify-center rounded-full bg-white text-[#030712] shadow-[0_18px_55px_rgba(255,255,255,0.16)] transition hover:scale-105 hover:bg-[#22d3ee] active:scale-95 md:h-16 md:w-16"
                  aria-label={isPlaying ? "Pause" : "Play"}
                >
                  {isPlaying ? <Pause className="h-6 w-6 fill-[#030712]" /> : <Play className="h-6 w-6 fill-[#030712]" />}
                </button>

                <div className="grid min-w-0 gap-3">
                  <div className="hidden h-8 items-end gap-1 md:flex">
                    {visualizerHeights.slice(0, 26).map((height, index) => (
                      <span
                        key={index}
                        className={`flex-1 rounded-full bg-gradient-to-t from-[#6366f1]/60 to-[#22d3ee]/90 ${isPlaying ? "wave-bar" : ""}`}
                        style={{
                          height: `${Math.max(8, height * 0.62)}px`,
                          animationDelay: `${index * 0.045}s`,
                          animationPlayState: isPlaying ? "running" : "paused",
                        }}
                      />
                    ))}
                  </div>

                  <div className="grid gap-2">
                    <div className={`flex items-center justify-between font-mono text-[10px] font-bold text-slate-500 ${isRtl ? "flex-row-reverse" : "flex-row"}`}>
                      <span>{formatTime(currentTime)}</span>
                      <span>{duration > 0 ? formatTime(duration) : currentEpisode.duration}</span>
                    </div>

                    <div
                      ref={progressBarRef}
                      role="slider"
                      aria-valuemin={0}
                      aria-valuemax={100}
                      aria-valuenow={Math.round(progress)}
                      tabIndex={0}
                      onPointerDown={handlePointerDown}
                      onPointerMove={handlePointerMove}
                      onPointerUp={handlePointerUp}
                      onPointerCancel={handlePointerUp}
                      className="group flex h-8 cursor-pointer touch-none items-center rounded-full outline-none focus-visible:ring-2 focus-visible:ring-[#22d3ee]/70"
                    >
                      <div className="relative h-2 w-full overflow-hidden rounded-full bg-white/10 transition-all duration-300 group-hover:h-3">
                        <div
                          className="absolute inset-y-0 left-0 rounded-full bg-gradient-to-r from-[#6366f1] via-[#a78bfa] to-[#22d3ee] shadow-[0_0_28px_rgba(34,211,238,0.55)]"
                          style={{ width: `${progress}%` }}
                        />
                        <div
                          className="absolute top-1/2 h-4 w-4 -translate-y-1/2 rounded-full border-2 border-white bg-[#22d3ee] opacity-0 shadow-[0_0_24px_rgba(34,211,238,0.9)] transition group-hover:opacity-100"
                          style={{ left: `calc(${progress}% - 8px)` }}
                        />
                      </div>
                    </div>
                  </div>
                </div>
              </div>
            </div>
          </motion.aside>
        )}
      </AnimatePresence>
    </main>
  );
}
